138 static bool ComboFilter(const char* id, char* buffer, int bufferlen, std::vector<std::string> hints, int num_hints, ComboFilterState& s)
139 {
140  struct fuzzy
141  {
142  static int score(const char* str1, const char* str2)
143  {
144  int score = 0, consecutive = 0, maxerrors = 0;
145  while (*str1 && *str2)
146  {
147  int is_leading = (*str1 & 64) && !(str1[1] & 64);
148  if ((*str1 & ~32) == (*str2 & ~32))
149  {
150  int had_separator = (str1[-1] <= 32);
151  int x = had_separator || is_leading ? 10 : consecutive * 5;
152  consecutive = 1;
153  score += x;
154  ++str2;
155  }
156  else
157  {
158  int x = -1, y = is_leading * -3;
159  consecutive = 0;
160  score += x;
161  maxerrors += y;
162  }
163  ++str1;
164  }
165  return score + (maxerrors < -9 ? -9 : maxerrors);
166  }
167  static int search(const char* str, int num, std::vector<std::string> words)
168  {
169  int scoremax = 0;
170  int best = -1;
171  for (int i = 0; i < num; ++i)
172  {
173  int score = fuzzy::score(words[i].c_str(), str);
174  int record = (score >= scoremax);
175  int draw = (score == scoremax);
176  if (record)
177  {
178  scoremax = score;
179  if (!draw) best = i;
180  else best = best >= 0 && strlen(words[best].c_str()) < strlen(words[i].c_str()) ? best : i;
181  }
182  }
183  return best;
184  }
185  };
186  using namespace ImGui;
187  bool done = InputText((std::string(id)).c_str(), buffer, bufferlen, ImGuiInputTextFlags_EnterReturnsTrue);
188  bool hot = s.activeIdx >= 0 && strcmp(buffer, hints[s.activeIdx].c_str());
189  if (hot)
190  {
191  int new_idx = fuzzy::search(buffer, num_hints, hints);
192  int idx = new_idx >= 0 ? new_idx : s.activeIdx;
193  s.selectionChanged = s.activeIdx != idx;
194  s.activeIdx = idx;
195  if (done || ComboFilter__DrawPopup(s, idx, hints, num_hints, id))
196  {
197  int i = s.activeIdx;
198  if (i >= 0)
199  {
200  strcpy(buffer, hints[i].c_str());
201  done = true;
202  }
203  }
204  }
205  return done;
206 }
